initiate stopped(Vessel)=true if
	stop_start(Vessel),
	happens coord(Vessel, Lon, Lat),
	nearPorts(Lon,Lat,[]).

terminate stopped(Vessel)=true if
	stop_end(Vessel).

initiate lowSpeed(Vessel)=true if
	slow_motion_start(Vessel),
	happens coord(Vessel, Lon, Lat),
	nearPorts(Lon,Lat,[]).

terminate lowSpeed(Vessel)=true if
	slow_motion_end(Vessel)
or	start stopped(Vessel)=true.

initiate withinArea(Vessel, AreaName)=true if
	isInArea(Vessel, AreaName).

terminate withinArea(Vessel, AreaName)=true if
	leavesArea(Vessel, AreaName).

initiate sailing(Vessel)=true if
	velocity(Vessel, Speed, Heading),
	Speed > 2.0.

terminate sailing(Vessel)=true if
	velocity(Vessel, Speed, Heading),
	Speed < 2.0
or	gap_start(Vessel).

initiate highSpeedIn(Vessel, AreaName)=true if
	isInArea(Vessel, AreaName),
	happens velocity(Vessel, Speed, Heading),
	speedArea(AreaName,SpeedArea),
	Speed > SpeedArea.

terminate highSpeedIn(Vessel, AreaName)=true if
	isInArea(Vessel, AreaName),
	happens velocity(Vessel, Speed, Heading),
	speedArea(AreaName,SpeedArea),
	Speed < SpeedArea
or	leavesArea(Vessel, AreaName)
or	gap_start(Vessel).

loitering(Vessel)=true iff
	((lowSpeed(Vessel)=true or stopped(Vessel)=true),
	withinArea(Vessel, AreaName)=true),
	duration > 600.

rendezVouz(Vessel1, Vessel2)=true iff
	(proximity(Vessel1, Vessel2)=true,
	(lowSpeed(Vessel1)=true or stopped(Vessel1)=true),
	(lowSpeed(Vessel2)=true or stopped(Vessel2)=true)),
	duration > 600.
